# 仅用递归函数和栈操作逆序一个栈

```c++
#include <stack>
#include <iostream>

using namespace std;

int getAndRemoveLastElement(stack<int>& stack) {
  int result = stack.top();
  stack.pop();
  if (stack.empty()) {
    return result;
  } else {
    int last = getAndRemoveLastElement(stack);
      stack.push(result);
      return last;
  }
}

void reverse(stack<int>& stack) {
  if (stack.empty()) {
    return ;
  }
  int ans = getAndRemoveLastElement(stack);
  reverse(stack);
  stack.push(ans);
}
int main()
{
  int n, m;
  stack<int> s1;
  cin >> n;
  for (int i = 0; i < n; i++) {
    cin >> m;
    s1.push(m);
  }
 // reverse(s1);
  while(!s1.empty()) {
    int s = s1.top();
    s1.pop();
    cout << s << " ";
  }
  return 0;
}
```
