# **根据后序数组重建搜索二叉树**

```C++
#include<iostream>
 using namespace std;
 
 int a[500007];
 
 bool isLRN(int a[],int start,int end)
 {
 	if(start==end)
 	  return true;
 	int less=-1,more=end;
 	for(int i=start;i<end;i++)
	 {
	 	if(a[i]<a[end])
	 	   less=i;
	 	else
	 	   more= more==end? i:more; 
	 } 
	 
	 if(less==-1 || more==end )
	 {
	 	return isLRN(a,start,end-1);
	 }
	 if(less!=more-1)
	   return false;
	 else
	    return isLRN(a,start, less) && isLRN(a,more,end-1);
 }
 
 int main()
 {
 	int n;
 	scanf("%d",&n);
 	for(int i=0;i<n;i++)
 	  scanf("%d",&a[i]);
 	bool k=isLRN(a,0,n-1);
 	if(k)
 	  cout<<"true"<<endl;
 	else
 	   cout<<"false"<<endl;
 	
 	return 0;
 } 
```
