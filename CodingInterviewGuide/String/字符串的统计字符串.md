# **字符串的统计字符串**

```c++
#include<iostream>
#include<vector>
#include<unordered_map>
using namespace std;

int main(){
    string s;
    while(cin>>s){
        char c = s[0];
        int n = 1;
        if(s.length()==1){
            cout<<s<<"_"<<1<<endl;
            
        }else{
            for(int i=1;i<s.length()-1;i++){
            if(s[i]==c){
                n++;
            }else{
                cout<<c<<"_"<<n<<"_";
                c=s[i];
                n=1;
            }
            
        }
        
            if(s[s.length()-1]==c){
                cout<<c<<"_"<<n+1<<endl;
            }
        else{
            cout<<c<<"_"<<n<<"_"<<s[s.length()-1]<<"_"<<1<<endl;
        }
        }
        
    }
}
```
