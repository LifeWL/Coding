# **删除多余的字符得到字典序最小的字符串**

```C++
#include<vector>
#include<iostream>
#include<string>
#include<set>

using namespace std;

class Solution{
public:
    string function(string& s){
        set<char> set_s;
        vector<int> vec_s(s.size(),0);
        for(auto& s1:s){
            set_s.insert(s1);
        }
        int begin = 0;
        for(auto& set_i:set_s){
            for(int j=begin;;++j){
                if(j==s.size()){
                    j = 0;
                }
                if(s[j] == set_i){
                    vec_s[j] = 1;
                    begin = j+1;
                    break;
                }
            }
        }
        string result = "";
        for(int i=0;i<s.size();++i){
            if(vec_s[i] == 1){
                result += s[i];
            }
        }
        cout << "result:" << result << endl;
        return result;
    }
};

int main(){
    string s = "dcbacbc";
    Solution obj_s;
    cout << obj_s.function(s) << endl;
    return 0;
}
```
