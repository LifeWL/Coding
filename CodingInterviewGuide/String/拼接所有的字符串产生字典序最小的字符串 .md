# 拼接所有的字符串产生字典序最小的字符串

```c++
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
    int n;
    cin>>n;
    string tmp;
    vector<string> arr;
    for(int i=0;i<n;++i)
    {
        cin>>tmp;
        arr.emplace_back(tmp);
    }
    sort(arr.begin(),arr.end(),[](string& a,string& b)
         {
            return a+b<b+a; 
         });
    for(auto s:arr)
    {
        cout<<s;
    }
    return 0;
}
```
