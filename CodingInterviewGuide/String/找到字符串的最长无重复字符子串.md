# **找到字符串的最长无重复字符子串**

```C++
#include <iostream>
using namespace std;

#define MAXVAL 1000010

int arr[MAXVAL];

int main()
{
    int n;
    scanf("%d",&n);
    int pre = 0;
    int ret = 0;
    int val;
    for (int i = 0;i<n;++i)
    {
        scanf("%d",&val);
        if (!arr[val] || arr[val]<pre)
        {
            if (i-pre+1>ret)
            {
                ret = i - pre + 1;
            }            
        }
        else
        {
            pre = arr[val];
        }
        arr[val] = i+1;
    }
    printf("%d",ret);
    return 0;
}
```
