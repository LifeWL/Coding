# **字符串的交错组成**

```c++
#include <iostream>
#include <string>

using namespace std;

int main()
{
    string str1,str2,str3;
    cin >> str1 >> str2 >> str3;
    
    int str1_len = str1.length();
    int str2_len = str2.length();
    int str3_len = str3.length();
    
    if(str1_len + str2_len != str3_len)
    {
        cout << "NO";
        return 0;
    }
    
    int i=0,j=0,k=0;
    
    while(i < str1_len && k < str3_len)
    {
        if(str1[i] == str3[k])
        {
            i++;
            k++;
        }
        else
            k++;
    }
    if(i != str1_len)
    {
        cout << "NO";
        return 0;
    }
    
    k=0;
    while(j < str2_len && k < str3_len)
    {
        if(str2[j] == str3[k])
        {
            j++;
            k++;
        }
        else
            k++;
    }
    if(j != str2_len)
    {
        cout << "NO";
        return 0;
    }
    
    cout << "YES";
    
    return 0;
}
```
