# 环形单链表的约瑟夫问题

```c++
#include <iostream>
using namespace std;

struct Node {
  int value;
  Node *next;
};

Node *josephusKill1(Node* head, int m) {
  if ((head == NULL) || (head->next == head) || (m < 1)) {
  return head;
} 
  Node *last = head;
  while (last->next != head) {
    last = last->next;
  }
  int count = 0;
  while (head != last) {
    if (++count == m) {
      last->next = head->next;
      count = 0;
    } else {
      last = last->next;
    }
    head = last->next;
  }
  return head;
}

int main() 
{
  int n, m;
  cin >> n >> m;
  Node *head = NULL;
  Node *p = NULL;
  for (int i = 1; i <= n; i++) {
    if (head == NULL) {
      head = new Node;
      head->value = i;
      head->next = NULL;
      p = head;
      continue;
    }
    p->next = new Node;
    p = p->next;
    p->value = i;
    p->next = NULL;
  }
  p->next = head;
  Node *node1 = josephusKill1(head, m);
  cout << node1->value << endl;
  return 0;
}
```
