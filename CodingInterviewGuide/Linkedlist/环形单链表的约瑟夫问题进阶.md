# 环形单链表的约瑟夫问题进阶

```c++
#include <iostream>
using namespace std;

struct Node {
  int value;
  Node *next;
};


Node *josephusKill2(Node *head, int m, int n) 
{
  if ((head == NULL) || (head->next == head) || (m < 1)) {
    return head;
  }
  Node *cur = head->next;
  int tmp = 1; 
  for(int i=2;i<=n;i++)
   {
       tmp=(tmp + m - 1) % i + 1;
   }
  while(--tmp != 0) {
    head = head->next;
  }
  head->next = head;
  return head;
}

int main() 
{
  int n, m;
  cin >> n >> m;
  Node *head = NULL;
  Node *p = NULL;
  for (int i = 1; i <= n; i++) {
    if (head == NULL) {
      head = new Node;
      head->value = i;
      head->next = NULL;
      p = head;
      continue;
    }
    p->next = new Node;
    p = p->next;
    p->value = i;
    p->next = NULL;
  }
  p->next = head;
  Node *node1 = josephusKill2(head, m, n);
  cout << node1->value << endl;
  return 0;
}
```
