# 判断一个链表是否为回文结构

```c++
# include <bits/stdc++.h>
using namespace std;

struct list_node{
    int val;
    struct list_node * next;
};

list_node * input_list(void)
{
    int n, val;
    list_node * phead = new list_node();
    list_node * cur_pnode = phead;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &val);
        if (i == 1) {
            cur_pnode->val = val;
            cur_pnode->next = NULL;
        }
        else {
            list_node * new_pnode = new list_node();
            new_pnode->val = val;
            new_pnode->next = NULL;
            cur_pnode->next = new_pnode;
            cur_pnode = new_pnode;
        }
    }
    return phead;
}


list_node* check(list_node * head)
{
    //////在下面完成代码
  stack<list_node*> sta;
  list_node *cur = head;
  while (cur != NULL) {
    sta.push(cur);
    cur = cur->next;
  }
  while (head != NULL) {
    if (head->val != (sta.top())->val) {
      cout << "false" <<endl;
      return head;
    }
    sta.pop();
    head = head->next;
  }
  cout << "true" <<endl;
  return head;
}


int main ()
{
    int L, R;
    list_node * head = input_list();
    check(head);
    return 0;
}
```

```c++
        }
    }

}


list_node * check(list_node * head)
{
    //////在下面完成代码
  if (head == NULL || head->next == NULL) {
    return head;
  }
  list_node *right = head->next;
  list_node *cur = head;
  while (cur->next != NULL && cur->next->next != NULL) {
    right = right->next;
    cur = cur->next->next;
  }
  stack<list_node *> sta;
  while (right != NULL) {
    sta.push(right);
    right = right->next;
  }
  while (!sta.empty()) {
    if (head->val != (sta.top())->val) {
      cout << "false" << endl;
      return head;
    }
    sta.pop();
    head = head->next;
  }
  cout << "true" << endl;
  return head;
}


int main ()
{
    int L, R;
    list_node * head = input_list();
    check(head);
    return 0;
```
